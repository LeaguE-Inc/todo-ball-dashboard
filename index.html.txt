    <script>
        const DEFAULT_TASKS = [

                id: 1,
                title: 'クライアント提案資料の完成',

            },
            {

            },
            {

                deadline: '2026-05-08',

                progress: 0,
                ballOwner: '営業部・A社',
                section: 'waiting'
            },
            {
                id: 5,
                title: 'ベンダーシステム導入スケジュール確認',
                description: '技術仕様書の納品待ち',
                deadline: '2026-05-15',
                priority: 'medium',
                ball: 'other',
                progress: 0,
                ballOwner: 'ITベンダー',
                section: 'waiting'
            },
            {
                id: 6,
                title: '社内ドキュメント整理',
                description: '過去プロジェクトのファイル整理',
                deadline: '2026-05-20',
                priority: 'low',
                ball: 'me',
                progress: 10,
                section: 'low'
            }
        ];

        class TodoBallApp {
            constructor() {
                this.tasks = this.loadTasks();
                this.boards = [
                    { id: 'high', label: '🔴 優先度: 高', class: 'high' },
                    { id: 'medium', label: '🟡 優先度: 中', class: 'medium' },
                    { id: 'low', label: '🟢 優先度: 低', class: 'low' },
                    { id: 'waiting', label: '⏳ 対応待ち', class: 'waiting' },
                    { id: 'reminder', label: '📢 リマインド', class: 'reminder' }
                ];
                this.init();
            }

            init() {
                this.setupDate();
                this.render();
                this.setupEventListeners();
                this.updateStats();
            }

            loadTasks() {
                const saved = localStorage.getItem('todoBallTasks');
                return saved ? JSON.parse(saved) : DEFAULT_TASKS;
            }

            saveTasks() {
                localStorage.setItem('todoBallTasks', JSON.stringify(this.tasks));
            }

            setupDate() {
                const today = new Date();
                const dateStr = today.toLocaleDateString('ja-JP', {
                    weekday: 'long',
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                });
                document.getElementById('today').textContent = dateStr;
            }

            getDaysUntil(deadline) {
                const today = new Date();
                today.setHours(0, 0, 0, 0);
                const deadlineDate = new Date(deadline);
                deadlineDate.setHours(0, 0, 0, 0);
                const diff = deadlineDate - today;
                return Math.ceil(diff / (1000 * 60 * 60 * 24));
            }

            getCountdownClass(days) {
                if (days < 0) return 'overdue';
                if (days === 0) return 'urgent';
                if (days <= 3) return 'urgent';
                if (days <= 7) return 'soon';
                return 'ok';
            }

            getCountdownText(days) {
                if (days < 0) return `⚠️ ${Math.abs(days)}日超過`;
                if (days === 0) return '🔴 本日期限';
                if (days === 1) return '⚠️ 明日期限';
                return `⏰ あと${days}日`;
            }

            getTasksBySection(sectionId) {
                return this.tasks.filter(t => t.section === sectionId);
            }

            render() {
                const container = document.getElementById('boardsContainer');
                container.innerHTML = '';

                for (const board of this.boards) {
                    const tasks = this.getTasksBySection(board.id);
                    const boardEl = this.createBoardElement(board, tasks);
                    container.appendChild(boardEl);
                }
            }

            createBoardElement(board, tasks) {
                const boardDiv = document.createElement('div');
                boardDiv.className = `board ${board.class}`;
                boardDiv.id = `board-${board.id}`;

                const title = document.createElement('h2');
                title.textContent = board.label;
                boardDiv.appendChild(title);

                const tasksList = document.createElement('div');
                tasksList.className = 'tasks-list';
                tasksList.id = `tasks-${board.id}`;

                if (tasks.length === 0) {
                    const empty = document.createElement('div');
                    empty.className = 'empty-state';
                    empty.innerHTML = '<p>✨ タスクなし</p>';
                    tasksList.appendChild(empty);
                } else {
                    for (const task of tasks) {
                        const taskEl = this.createTaskElement(task);
                        tasksList.appendChild(taskEl);
                    }
                }

                boardDiv.appendChild(tasksList);
                return boardDiv;
            }

            createTaskElement(task) {
                const taskDiv = document.createElement('div');
                taskDiv.className = `task-card priority-${task.priority} ball-${task.ball}`;
                taskDiv.id = `task-${task.id}`;
                taskDiv.draggable = true;

                const days = this.getDaysUntil(task.deadline);
                const countdownClass = this.getCountdownClass(days);
                const countdownText = this.getCountdownText(days);

                taskDiv.innerHTML = `
                    <div class="task-header">
                        <div class="task-title">${task.title}</div>
                        <span class="task-priority ${task.priority}">${task.priority === 'high' ? '高' : task.priority === 'medium' ? '中' : '低'}</span>
                    </div>
                    <div class="task-body">
                        <div class="task-description">${task.description}</div>
                        <div class="progress-bar">
                            <div class="progress-fill" style="width: ${task.progress}%"></div>
                        </div>
                        <div class="progress-text">${task.progress}% 完了</div>
                    </div>
                    <div class="task-meta">
                        <div class="deadline">
                            <span>📅 ${task.deadline}</span>
                            <span class="countdown ${countdownClass}">${countdownText}</span>
                        </div>
                        <div class="ball-indicator ${task.ball === 'other' ? 'waiting' : ''}" data-task-id="${task.id}">
                            ${task.ball === 'me' ? '🤖' : '⏳'}
                        </div>
                    </div>
                `;

                taskDiv.addEventListener('dragstart', (e) => {
                    e.dataTransfer.effectAllowed = 'move';
                    e.dataTransfer.setData('taskId', task.id);
                    taskDiv.classList.add('dragging');
                });

                taskDiv.addEventListener('dragend', () => {
                    taskDiv.classList.remove('dragging');
                });

                const ballIndicator = taskDiv.querySelector('.ball-indicator');
                ballIndicator.addEventListener('click', (e) => {
                    e.stopPropagation();
                    this.showBallMenu(task, ballIndicator);
                });

                taskDiv.addEventListener('contextmenu', (e) => {
                    e.preventDefault();
                    if (confirm(`"${task.title}" を削除しますか？`)) {
                        this.deleteTask(task.id);
                    }
                });

                return taskDiv;
            }

            showBallMenu(task, element) {
                const menu = document.createElement('div');
                menu.className = 'ball-menu show';

                const options = [
                    { label: '🤖 自分', value: 'me' },
                    { label: '⏳ 営業部', value: 'sales' },
                    { label: '⏳ IT部', value: 'it' },
                    { label: '⏳ その他', value: 'other' }
                ];

                for (const option of options) {
                    const item = document.createElement('div');
                    item.className = 'ball-menu-item';
                    item.textContent = option.label;
                    item.addEventListener('click', () => {
                        this.updateTaskBall(task.id, option.value);
                        menu.remove();
                    });
                    menu.appendChild(item);
                }

                element.parentElement.appendChild(menu);
                setTimeout(() => menu.addEventListener('click', (e) => e.stopPropagation()), 0);
            }

            updateTaskBall(taskId, newBall) {
                const task = this.tasks.find(t => t.id === taskId);
                if (!task) return;

                task.ball = newBall;

                if (newBall === 'me') {
                    task.section = task.priority;
                } else {
                    task.section = 'waiting';
                }

                this.saveTasks();
                this.render();
                this.updateStats();
            }

            deleteTask(taskId) {
                this.tasks = this.tasks.filter(t => t.id !== taskId);
                this.saveTasks();
                this.render();
                this.updateStats();
            }

            addTask(title, description, deadline, priority) {
                const newId = Math.max(...this.tasks.map(t => t.id), 0) + 1;
                const newTask = {
                    id: newId,
                    title,
                    description,
                    deadline,
                    priority,
                    ball: 'me',
                    progress: 0,
                    section: priority
                };
                this.tasks.push(newTask);
                this.saveTasks();
                this.render();
                this.updateStats();
            }

            updateStats() {
                const totalTasks = this.tasks.length;
                const myTasks = this.tasks.filter(t => t.ball === 'me').length;
                const waitingTasks = this.tasks.filter(t => t.ball !== 'me').length;
                const urgentTasks = this.tasks.filter(t => {
                    const days = this.getDaysUntil(t.deadline);
                    return days <= 3 && days >= 0;
                }).length;

                document.getElementById('totalTasks').textContent = totalTasks;
                document.getElementById('myTasks').textContent = myTasks;
                document.getElementById('waitingTasks').textContent = waitingTasks;
                document.getElementById('urgentTasks').textContent = urgentTasks;
            }

            setupEventListeners() {
                for (const board of this.boards) {
                    const tasksList = document.getElementById(`tasks-${board.id}`);
                    if (tasksList) {
                        tasksList.addEventListener('dragover', (e) => {
                            e.preventDefault();
                            e.dataTransfer.dropEffect = 'move';
                            tasksList.style.opacity = '0.8';
                        });

                        tasksList.addEventListener('dragleave', () => {
                            tasksList.style.opacity = '1';
                        });

                        tasksList.addEventListener('drop', (e) => {
                            e.preventDefault();
                            tasksList.style.opacity = '1';
                            const taskId = parseInt(e.dataTransfer.getData('taskId'));
                            const task = this.tasks.find(t => t.id === taskId);
                            if (task) {
                                task.section = board.id;
                                this.saveTasks();
                                this.render();
                                this.updateStats();
                            }
                        });
                    }
                }

                document.getElementById('addTaskBtn').addEventListener('click', () => {
                    const title = prompt('タスク名を入力:');
                    if (!title) return;
                    const description = prompt('説明を入力（オプション）:') || '';
                    const deadline = prompt('期限を入力 (YYYY-MM-DD):');
                    if (!deadline) return;
                    const priority = confirm('優先度を高にしますか？\nOK=高, キャンセル=中') ? 'high' : 'medium';

                    this.addTask(title, description, deadline, priority);
                });
            }
        }

        const app = new TodoBallApp();
    </script>
</body>
